PREFIX : <urn:absolute/prototyp#>
PREFIX owl: <http://www.w3.org/2002/07/owl#>

SELECT ?in_ist_neu ?feldstaerke ?feldEinheit ?paramName ?wert ?einheit ?out_ist_neu ?modelName

WHERE {

#* update_new
        # Kommando um neue Parameter einzufügen.


?p1 a :Streckung;
        :hat_Wert ?streckung.
FILTER(?streckung = 1)
?p2 a :elektrische_Feldstaerke;
          :hat_Wert ?feldstaerke;
          :hat_Einheit ?feldEinheit;
          :ist_Neu ?in_ist_neu;
          :korrespondiert_mit ?p1.
        # Über :korrespondiert_mit wird angegeben, welche Parameter zueinander in Bezug stehen sollen.
FILTER(?feldstaerke IN (35000000, 36000001))
        # Filter können über = oder über IN definiert werden, falls mehrere Werte berechnet werden sollen.
        # Bei Angaben mehrerer Werte wird entsprechend jede mögliche Kombination zu den Korrespondenzen gebildet.

#* update_end
        # Ab hier werden keine neuen Parameter erkannt.

        # Abfrage von der Berechnung wie gewohnt
?model :hat_Eingabe ?p2;
       :hat_Name ?modelName.

?pcalc :hat_Herkunft ?model;
       :ist_Neu ?out_ist_neu;
       :hat_Name ?paramName;
       :hat_Wert ?wert;
       :hat_Einheit ?einheit.


?mat :hat_Parameter ?p1;
     :hat_Parameter ?p2;
         :hat_Name ?materialName.
FILTER (?materialName = 'Stapelaktor_1')
        # die letzten beide Teile gehören noch zu dem Eintragen von Parametern, damit der Parameter weiß zu welchem Material
	#  er gehört. Hier wird kein update_new benötigt, da das Material nicht neu ist.
}
